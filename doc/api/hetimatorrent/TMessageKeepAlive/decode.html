static Future<TMessageKeepAlive> decode(EasyParser parser) async {
  Completer c = new Completer();
  TMessageKeepAlive message = new TMessageKeepAlive();
  parser.push();
  try {
    int size = await parser.readInt(ByteOrder.BYTEORDER_BIG_ENDIAN);
    if (size != 0) {
      throw {};
    }
    parser.pop();
    return message;
  } catch (e) {
    parser.back();
    parser.pop();
    throw e;
  }
  return c.future;
}
