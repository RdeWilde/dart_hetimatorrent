TorrentClientPeerInfo getPeerInfoFromId(int id) {
  for (int i = 0; i < peerInfos.length; i++) {
    TorrentClientPeerInfo info = peerInfos.getSequential(i);
    if (info.id == id) {
      return info;
    }
  }
  return null;
}
