static String toText(Object oo, String key, String def) {
  try {
    if (!(oo is Map)) {
      return def;
    }
    Map p = oo as Map;
    if (!p.containsKey(key)) {
      return def;
    }
    if (!(p[key] is data.Uint8List)) {
      return def;
    }
    return convert.UTF8.decode((p[key] as data.Uint8List).toList());
  } catch (e) {
    return def;
  }
}
