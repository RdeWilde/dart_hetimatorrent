Future stop() async {
  await this._torrent.stop();
  isGo = false;
  if (useDht == true) {
    await _dht.stopSearchPeer(new KId(_tracker.infoHash)).catchError((e) {});
  }
  return await _startTracker(0).catchError((e) {});
}
