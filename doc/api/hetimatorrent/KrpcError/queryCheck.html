static bool queryCheck(Map<String, Object> messageAsMap) {
  if (!messageAsMap.containsKey("e")) {
    return false;
  }
  Object e = messageAsMap["e"];
  if (!(e is List) || (e as List).length < 2 || !messageAsMap.containsKey("t") || !messageAsMap.containsKey("y")) {
    return false;
  }
  return true;
}
