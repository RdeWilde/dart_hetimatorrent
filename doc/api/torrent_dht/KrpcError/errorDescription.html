static String errorDescription(int errorCode) {
  switch (errorCode) {
    case 201:
      return "Generic Error";
    case 202:
      return "Server Error";
    case 203:
      return "Protocol Error, such as a malformed packet, invalid arguments, or bad token";
    case 204:
      return "Method Unknown";
    default:
      return "Unknown";
  }
}
