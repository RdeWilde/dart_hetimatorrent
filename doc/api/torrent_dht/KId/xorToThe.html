KId xorToThe(int x, {KId output: null, bool repeat: false}) {
  if (output == null) {
    output = new KId.zeroClear();
  }

  for (int i = 0; i < this._values.length; i++) {
    output._values[i] = this._values[i];
  }

  if (x == 0) {
    return output;
  } else {
    x -= 1;
    int i = x ~/ 8;
    int v = x % 8;
    int d = 0;
    if (repeat) {
      for (int j = 0; j <= v; j++) {
        d |= (0x01 << j);
      }
    } else {
      d = (0x01 << v);
    }
    output._values[19 - i] = this._values[19 - i] ^ d;
    return output;
  }
}
