Future<Map> decodeDictionElements(hetima.EasyParser parser) async {
  Map ret = new Map();
  while (true) {
    String key = await decodeString(parser);
    ret[key] = await decodeBenObject(parser);
    List<int> v = await parser.getPeek(1);
    if (v[0] == 0x65) {
      //e
      return ret;
    }
  }
}
