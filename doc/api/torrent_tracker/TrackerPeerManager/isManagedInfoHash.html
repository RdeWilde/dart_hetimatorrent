bool isManagedInfoHash(List<int> infoHash) {
  if (infoHash == null) {
    return false;
  }
  if (_managdInfoHash.length != infoHash.length) {
    return false;
  }
  for (int i = 0; i < _managdInfoHash.length; i++) {
    if (infoHash[i] != _managdInfoHash[i]) {
      return false;
    }
  }
  return true;
}
